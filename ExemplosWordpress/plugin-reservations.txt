// pega o evento quando um post é adicionado
add_action('save_post_rtb-booking', 'san_save_reservations_in_custom_table');

// pega o evento quando o plugin é ativo
register_activation_hook(__FILE__, 'san_register_custom_table_reservations');

//adiciona a tabela
function san_register_custom_table_reservations(){
    global $wpdb;

    $wpdb->query("
    CREATE TABLE reservations( 
        id int not null primary key auto_increment, 
        date varchar(11) not null, 
        time varchar(10) not null, 
        party varchar(12) not null, 
        name varchar(255) not null, 
        email varchar(255) not null, 
        phone varchar(20) not null)
    ");
}

// insert dentro da tabela, quando o evento de save do post type rtb-booking
function san_save_reservations_in_custom_table(){
    global $wpdb;

    $wpdb->insert('reservations', [
        'date' => $_POST['rtb-date'],
        'time' => $_POST['rtb-time'],
        'party' => $_POST['rtb-party'],
        'name' => $_POST['rtb-name'],
        'email' => $_POST['rtb-email'],
        'phone' => $_POST['rtb-phone'],
    ]);
}
